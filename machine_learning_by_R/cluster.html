wssplot<-function(data,nc=5,seed=1234){
  wss<-(nrow(data)-1)*sum(apply(data,2,var))
  for(i in 1:nc){
    set.seed(seed)
    wss[i]<-sum(kmeans(data,centers=i)$withinss)
  }
  plot(1:nc,wss,type="b",xlab="Number of Clusters",ylab="Within groups sum of squares")}
wssplot(dataset)
